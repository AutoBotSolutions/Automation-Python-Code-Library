def log_message(self, message):
        """Log messages to the textarea."""
        timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
        self.text_area.insert(tk.END, f"{timestamp} - {message}\n")
        self.text_area.yview(tk.END)
